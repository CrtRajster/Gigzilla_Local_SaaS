async function handleAppSumoWebhook(request, env) {
  const body = await request.json();

  // AppSumo sends:
  // {
  //   "email": "customer@example.com",
  //   "plan_id": "gigzilla_lifetime",
  //   "uuid": "license-key-here"
  // }

  // Create customer in Stripe with "lifetime" metadata
  const customer = await stripe.customers.create({
    email: body.email,
    metadata: {
      source: 'appsumo',
      license_type: 'lifetime',
      appsumo_uuid: body.uuid
    }
  });

  // Create a "subscription" that never expires
  const subscription = await stripe.subscriptions.create({
    customer: customer.id,
    items: [{ price: 'price_lifetime_internal' }],
    metadata: {
      plan: 'lifetime',
      source: 'appsumo'
    }
  });

  // Send license key via email

    Download Gigzilla: https://gigzilla.site/download
