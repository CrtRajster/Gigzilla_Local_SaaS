## ğŸ“Š Implementation Details

### Database Schema

```sql
-- User profile table
CREATE TABLE user_profile (
  id SERIAL PRIMARY KEY,
  name VARCHAR(255),
  business_name VARCHAR(255),
  email VARCHAR(255) UNIQUE,
  phone VARCHAR(50),
  whatsapp VARCHAR(50),
  photo_url TEXT,
  timezone VARCHAR(50),
  country VARCHAR(100),
  tax_id VARCHAR(100),
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- Payment details
CREATE TABLE payment_accounts (
  id SERIAL PRIMARY KEY,
  user_profile_id INTEGER REFERENCES user_profile(id),
  provider VARCHAR(50), -- paypal, stripe, bank
  account_email VARCHAR(255),
  account_identifier VARCHAR(255), -- IBAN, etc.
  oauth_token TEXT, -- encrypted
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Notification preferences
CREATE TABLE notification_preferences (
  id SERIAL PRIMARY KEY,
  user_profile_id INTEGER REFERENCES user_profile(id),
  channel VARCHAR(50), -- desktop, email, sms, whatsapp
  event_type VARCHAR(100), -- payment_received, invoice_overdue, etc.
  enabled BOOLEAN DEFAULT true,
  settings JSONB, -- custom settings per channel
  created_at TIMESTAMP DEFAULT NOW()
);

-- Connected services
CREATE TABLE connected_services (
  id SERIAL PRIMARY KEY,
  user_profile_id INTEGER REFERENCES user_profile(id),
  service_name VARCHAR(100), -- upwork, fiverr, gmail, etc.
  oauth_token TEXT, -- encrypted
  refresh_token TEXT, -- encrypted
  token_expires_at TIMESTAMP,
  is_active BOOLEAN DEFAULT true,
  metadata JSONB,
  last_synced_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW()
);
```
